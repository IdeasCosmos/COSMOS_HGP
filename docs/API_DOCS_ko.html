🐍 Python 클라이언트

from cosmos import CosmosAPI

# API 클라이언트 초기화
api = CosmosAPI(api_key="test_key_12345")

# 코돈 분석
result = api.analyze_codon("""
def fibonacci(n):
    if n <= 1:
        return n
    return fibonacci(n-1) + fibonacci(n-2)
""")
print(result)

# 속도 계산
velocity = api.calculate_velocity(layer=3, impact=0.26)
print(velocity)
